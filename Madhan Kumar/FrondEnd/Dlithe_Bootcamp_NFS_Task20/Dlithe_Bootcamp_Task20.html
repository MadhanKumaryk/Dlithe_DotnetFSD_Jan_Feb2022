Encapsulation

          
            function Animal()
            {
                this.AnimalName="AnimalName";
                throw new Error("You cannot create an instance of Abstract Class");
            }
            
         Animal.prototype.display=function()
         {
             return "Animal is: "+this.AnimalName;
         }
         //Creating a constructor function
         function Carnivoros(AnimalName)
         {
             this.AnimalName=AnimalName;
         }
         //Creating object without using the function constructor
         Carnivoros.prototype=Object.create(Animal.prototype);
     
         var Carnivoros=new Carnivoros("Lion");
         document.writeln(Carnivoros.display());

Encapsulation using class

          
            class Person
            {  
              constructor()  
              {  
                 var id;  
                 var dob;  
              }  
                getId()  
                  {  
                    return this.id;  
                  }  
                setId(id)  
                {  
                  this.id=id;  
                }  
                  
                getDob()  
                {  
                  return this.dob;  
                }  
              setDob(dob)  
              {  
                this.dob=dob;  
              }  
            
              }
