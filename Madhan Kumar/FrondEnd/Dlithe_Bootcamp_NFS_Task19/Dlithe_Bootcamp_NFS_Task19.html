            var Animal = function () {}
                Animal.prototype.select = function()
                {
                 return "Animals";
                }
                var lion = function() {}
                 lion.prototype = Object.create(Animal.prototype);
                 lion.prototype.select = function()               
                 {                                                        
                   return "lion";
                 }
                 var tiger = function(){}
                  tiger.prototype = Object.create(Animal.prototype);
                  tiger.prototype.select = function()
                  {
                      return "tiger";
                  }
                  var cheeta =function(){}
                  cheeta.prototype = Object.create(Animal.prototype);
                  cheeta.prototype.select=function()
                  {
                      return "cheeta";
              
                  }
                
                
                 var monkey = function() {}
                 monkey.prototype = Object.create(Animal.prototype); 
                 //shared parent property
                 var Animals = [new Animal(), new lion(), new tiger(), new cheeta(), new monkey()];
              
                 Animals.forEach(function(Animal)
                 {
                   document.write(Animal.select());
                   document.write("");
                 });

                function parent(A,B){
                    this.A=A;
                    this.B=B;
                    parent.prototype.getparent = function()
                    {
                        return A + ""+ B;
                    }
                   }
                   function child(A,B,C){
                       parent.call(this,A,B);
                       this.C=C;
                   }
                   child.prototype = new parent();
                   child.prototype.constructor = child;
                   var obj = new child("A","B","C");
                   document.write(obj.getparent());
